Perintah Dasar CLI Windows

TYPE "Nama File"	-> Perintah Untuk Membuka File
	Contoh
Type Sabarudin		-> Membuka file Sabarudin

COPY CON "Nama File"	-> Perintah Untuk Membuat File
	Contoh
COPY CON Sabarudin	-> Membuat File Sabarudin

DEL "Nama File"		-> Perintah Untuk Menghapus File
	Contoh
DEL Sabarudin		-> Menghapus File Sabarudin

CALL "Nama File"	-> Perintah Untuk Mengeksekusi File
	Contoh
CALL "Sabarudin.bat"	-> Mengeksekusi File Sabarudin

ECHO (Isi File) > "Nama File"	-> Perintah Untuk Membuat File Dan Mengisinya
	Contoh
ECHO Tugas 1 > Tugas-1.txt	-> Membuat File Dengan Nama Tugas-1.txt yang berisikan Tugas 1
